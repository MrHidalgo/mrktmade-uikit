                      function accordionCB() {
	if(!document.querySelector('.accordion__box-head')) return;

	const accordionHeadArr = document.querySelectorAll('.accordion__box-head');

	for(let i = 0; i < accordionHeadArr.length; i++) {
		accordionHeadArr[i].addEventListener('click', (ev) => {
			const el = ev.currentTarget;

			for(let j = 0; j < document.querySelectorAll('.accordion__box-body').length; j++) {
				document.querySelectorAll('.accordion__box-body')[j].style.display = 'none';
			}
			el.nextElementSibling.style.display = 'block';

			for(let k = 0; k < accordionHeadArr.length; k++) {
				accordionHeadArr[k].classList.remove('is-active');
			}
			el.classList.add('is-active');
		}, false);
	}
}
